\begin{align*} \fe{f}{x}&=\frac{\fe{\sin^5}{x}-\sqrt{\fe{\sin}{x}}}{\fe{\sin}{x}}\\ &=\fe{\sin^4}{x}-\left(\fe{\sin}{x}\right)^{-\sfrac{1}{2}}\\ \fe{\fd{f}}{x}&=4\fe{\sin^3}{x}\cdot\lzoo{x}{\fe{\sin}{x}}+\frac{1}{2}\left(\fe{\sin}{x}\right)^{-\sfrac{3}{2}}\cdot\lzoo{x}{\fe{\sin}{x}}\\ &=4\fe{\sin^3}{x}\cdot\fe{\cos}{x}+\frac{1}{2}\left(\fe{\sin}{x}\right)^{-\sfrac{3}{2}}\cdot\fe{\cos}{x}\\ &=\left(4\fe{\sin^3}{x}+\frac{1}{2\sqrt{\fe{\sin^3}{x}}}\right)\fe{\cos}{x} \end{align*}
