\begin{align*} \qquad\;m_{\text{tan}}&=\lim_{h\to0}\frac{\fe{f}{4+h}-\fe{f}{4}}{h}\\ &=\lim_{h\to0}\frac{\sqrt{4+h}-\sqrt{4}}{h}\\ &=\lim_{h\to0}\left(\frac{\sqrt{4+h}-2}{h}\cdot\frac{\sqrt{4+h}+2}{\sqrt{4+h}+2}\right)\\ &=\lim_{h\to0}\frac{4+h-4}{h\left(\sqrt{4+h}+2\right)}\\ &=\lim_{h\to0}\frac{h}{h\left(\sqrt{4+h}+2\right)}\\ &=\lim_{h\to0}\frac{1}{1\left(\sqrt{4+h}+2\right)}\\ &=\frac{1}{\sqrt{4+0}+2}\\ &=\frac{1}{4} \end{align*}
